// Список игр с указанием локального файла const GAMES = [ { id: "game_ferms", title: "Ферма-кликер", icon: "icon.png", file: "game.html" }, // если есть другие игры, добавьте их сюда: // { id: "race", title: "Гонки", icon: "icon.png", file: "index.html" }, // { id: "puzzle", title: "Пазл", icon: "icon.png", file: "index.html" } ]; // Запуск игры (локально, без GitHub) function launchGame(gameId) { // Находим игру по id const game = GAMES.find(g => g.id === gameId); if (!game) { console.error('Игра не найдена:', gameId); return; } // Скрываем меню, показываем контейнер document.getElementById('menu').style.display = 'none'; document.getElementById('gameContainer').style.display = 'block'; const frame = document.getElementById('gameFrame'); // Формируем локальный путь: папка_игры / файл_игры const localPath = `${gameId}/${game.file}`; frame.src = localPath; console.log(`Запускаем локальную игру: ${localPath}`); }
